Напиши программу приложение для управления задачами с модулем Tkinter для работы с PyCharm
внеси дополнения: кнопки имеют разные цвета + кнопку "выполнено"

Давайте добавим цвета для кнопок и добавим кнопку "Выполнено", которая будет перемещать задачу
в отдельный список выполненных задач.

import tkinter as tk
from tkinter import messagebox

class TaskManagerApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Task Manager")

        self.tasks = []
        self.completed_tasks = []

        self.frame = tk.Frame(self.root)
        self.frame.pack(pady=10)

        self.task_listbox = tk.Listbox(
            self.frame, width=50, height=10, bd=0
        )
        self.task_listbox.pack(side=tk.LEFT, fill=tk.BOTH)

        self.scrollbar = tk.Scrollbar(self.frame)
        self.scrollbar.pack(side=tk.RIGHT, fill=tk.BOTH)

        self.task_listbox.config(yscrollcommand=self.scrollbar.set)
        self.scrollbar.config(command=self.task_listbox.yview)

        self.entry = tk.Entry(self.root, width=52)
        self.entry.pack(pady=10)

        self.add_task_button = tk.Button(
            self.root, text="Добавить задачу", bg="lightgreen", command=self.add_task
        )
        self.add_task_button.pack(pady=5)

        self.delete_task_button = tk.Button(
            self.root, text="Удалить задачу", bg="lightcoral", command=self.delete_task
        )
        self.delete_task_button.pack(pady=5)

        self.complete_task_button = tk.Button(
            self.root, text="Выполнено", bg="lightblue", command=self.complete_task
        )
        self.complete_task_button.pack(pady=5)

        self.show_completed_button = tk.Button(
            self.root, text="Показать выполненные", bg="lightyellow", command=self.show_completed
        )
        self.show_completed_button.pack(pady=5)

    def add_task(self):
        task = self.entry.get()
        if task:
            self.tasks.append(task)
            self.update_task_list()
            self.entry.delete(0, tk.END)
        else:
            messagebox.showwarning("Предупреждение", "Введите описание задачи!")

    def delete_task(self):
        try:
            selected_task_index = self.task_listbox.curselection()[0]
            del self.tasks[selected_task_index]
            self.update_task_list()
        except IndexError:
            messagebox.showwarning("Предупреждение", "Выберите задачу для удаления!")

    def complete_task(self):
        try:
            selected_task_index = self.task_listbox.curselection()[0]
            task = self.tasks.pop(selected_task_index)
            self.completed_tasks.append(task)
            self.update_task_list()
            messagebox.showinfo("Информация", f"Задача '{task}' выполнена!")
        except IndexError:
            messagebox.showwarning("Предупреждение", "Выберите задачу для отметки как выполненную!")

    def show_completed(self):
        if not self.completed_tasks:
            messagebox.showinfo("Выполненные задачи", "Нет выполненных задач.")
        else:
            completed_tasks_str = "\n".join(self.completed_tasks)
            messagebox.showinfo("Выполненные задачи", f"Выполненные задачи:\n{completed_tasks_str}")

    def update_task_list(self):
        self.task_listbox.delete(0, tk.END)
        for task in self.tasks:
            self.task_listbox.insert(tk.END, task)

if __name__ == "__main__":
    root = tk.Tk()
    app = TaskManagerApp(root)
    root.mainloop()

Изменения
Кнопка "Добавить задачу" теперь зеленого цвета.
Кнопка "Удалить задачу" теперь красного цвета.
Добавлена кнопка "Выполнено": отмечает задачу как выполненную и перемещает её в список выполненных задач.
Добавлена кнопка "Показать выполненные": показывает выполненные задачи в отдельном окне.
